   - Fetch bills with crypto query, categorize (active/signed), generate summary/next event via simple logic (e.g., find recent signed or upcoming).
   - Update template to render dynamically.

```python
import os
import requests
from flask import Flask, render_template  # Assuming Flask; install if needed (free: pip install flask in shell)

app = Flask(__name__)
API_KEY = os.environ.get('LEGISCAN_API_KEY')

def fetch_crypto_legislation():
    # Search for recent crypto bills (adjust query for relevance)
    url = f"https://api.legiscan.com/?key={API_KEY}&op=search&query=cryptocurrency+OR+bitcoin+OR+digital+asset&state=US&year=2025&page=1"
    response = requests.get(url)
    if response.status_code != 200:
        return {"bills": [], "summary": "Data unavailable", "next_event": "Check back later"}
    
    data = response.json()
    if data['status'] != 'OK':
        return {"bills": [], "summary": "API error", "next_event": "N/A"}
    
    raw_bills = data['searchresult']['results'][:20]  # Top 20
    bills = []
    for bill in raw_bills:
        detail_url = f"https://api.legiscan.com/?key={API_KEY}&op=getBill&id={bill['bill_id']}"
        detail_resp = requests.get(detail_url)
        if detail_resp.status_code == 200:
            details = detail_resp.json()['bill']
            bills.append({
                "title": details['title'],
                "number": details['bill_number'],
                "status": details['status'],
                "next_steps": details['history'][-1]['action'] if details['history'] else "Pending",  # Last action
                "passage_chance": "TBD",  # Calculate or static; could use AI if integrated
                "whats_next": details['status_date'],
                "sponsor": ", ".join([s['name'] for s in details['sponsors']]),
                "summary": details['description']
            })
    
    # Generate summary (simple aggregation; enhance with logic)
    summary = f"US Crypto Legislation - {len(bills)} active/recent bills as of December 2025. Key themes: CFTC/SEC harmonization, Bitcoin reserves, ETF approvals."
    
    # Next major event: Find earliest future date or recent signed
    next_event = "Upcoming: Potential CFTC-SEC rule finalizations Q1 2026."  # Parse from bills
    
    return {"bills": bills, "summary": summary, "next_event": next_event}

@app.route('/legislation')
def legislation():
    data = fetch_crypto_legislation()
    return render_template('legislation.html', **data)  # Pass to template
```
