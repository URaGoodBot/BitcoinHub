To automate tracking in Replit (a fine sandbox for macro scripts, akin to TradingView but with Python's edge), I've crafted a starter `main.py` file. It fetches live data via the FRED API (free, but grab your key at [fred.stlouisfed.org](https://fred.stlouisfed.org/docs/api/api_key.html)â€”treat it like a vault code). Run `pip install fredapi pandas matplotlib` in Replit's shell first. The script pulls the latest observations, computes YoY growth (flagging >5% anomalies as "alerts"), saves to CSV for export/analysis, and plots M2 as a baseline chart. Tweak thresholds or add Pine-like indicators (e.g., EMAs via TA-Lib) as you scale.

Copy-paste this into Replit as `main.py` and hit runâ€” it'll prompt for your API key on first go.

```python
import os
import pandas as pd
from fredapi import Fred
import matplotlib.pyplot as plt
from datetime import datetime, timedelta

# Prompt for API key (or use env var for security)
api_key = input("Enter your FRED API key (or press Enter if set as FRED_API_KEY env var): ").strip() or os.getenv('FRED_API_KEY')
if not api_key:
    raise ValueError("API key required. Get one at https://fred.stlouisfed.org/docs/api/api_key.html")

fred = Fred(api_key=api_key)

# Core series: ID -> Human-readable name
series_dict = {
    'M2SL': 'M2 Money Supply',
    'M1SL': 'M1 Money Supply',
    'RRPONTSYD': 'Reverse Repo (Daily)',
    'WTREGEN': 'TGA (Weekly)',
    'WALCL': 'Fed Total Assets',
    'WRESBAL': 'Bank Reserves',
    'CURRCIR': 'Currency in Circulation',
    'BOGMBASE': 'Monetary Base'
}

data_dict = {}
alerts = []

print("Fetching latest data...\n")

for series_id, name in series_dict.items():
    try:
        # Fetch last 24 months for YoY calc (adjust as needed)
        end_date = datetime.now()
        start_date = end_date - timedelta(days=730)
        s = fred.get_series(series_id, start_date, end_date)
        
        if len(s) < 2:
            print(f"Insufficient data for {name}")
            continue
            
        latest = s.iloc[-1]
        latest_date = s.index[-1].strftime('%Y-%m-%d')
        prev_year = s.loc[s.index >= (s.index[-1] - timedelta(days=365))].iloc[0] if len(s.loc[s.index >= (s.index[-1] - timedelta(days=365))]) > 0 else latest
        
        yoy_growth = ((latest - prev_year) / prev_year * 100) if prev_year != 0 else 0
        data_dict[name] = s
        
        print(f"{name}: Latest = ${latest:,.0f}B (as of {latest_date}) | YoY Growth: {yoy_growth:.1f}%")
        if abs(yoy_growth) > 5:  # Threshold for "irregularity" alert (tune this)
            alerts.append(f"ALERT: {name} shows {yoy_growth:.1f}% YoY shiftâ€”potential distortion.")
            
    except Exception as e:
        print(f"Error fetching {series_id}: {e}")

if alerts:
    print(f"\nðŸš¨ Anomalies Detected:\n" + "\n".join(alerts))

# Save all data to CSV for Replit export/analysis
if data_dict:
    df = pd.DataFrame(data_dict)
    df.to_csv('fred_money_supply_tracker.csv')
    print(f"\nFull dataset (last 24 months) saved to 'fred_money_supply_tracker.csv'")

    # Quick plot: M2 as baseline (scale others similarly if desired)
    if 'M2 Money Supply' in data_dict:
        plt.figure(figsize=(12, 6))
        data_dict['M2 Money Supply'].plot(label='M2 ($B)', linewidth=2)
        plt.title('M2 Money Supply Tracker (Latest Data)', fontsize=14)
        plt.ylabel('Billions of USD')
        plt.xlabel('Date')
        plt.legend()
        plt.grid(True, alpha=0.3)
        plt.tight_layout()
        plt.savefig('m2_tracker.png')  # Save for Replit gallery
        plt.show()
        print("\nM2 plot generated and saved as 'm2_tracker.png'")

print("\nRun periodically via cron/scheduler in Replit for ongoing surveillance. Scale with alerts to email/Slack.")
```
