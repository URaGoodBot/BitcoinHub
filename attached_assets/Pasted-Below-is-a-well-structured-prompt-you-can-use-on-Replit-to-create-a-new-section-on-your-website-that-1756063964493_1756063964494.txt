```javascript
// script.js
const indicators = [
  { name: "RSI", description: "Measures the speed and change of price movements to identify overbought or oversold conditions.", sources: ["TradingView (tradingview.com)", "CoinGecko (coingecko.com)"] },
  // Add 29 more indicators
];

const indicatorsContainer = document.getElementById("indicators");
const searchInput = document.getElementById("search");

function renderIndicators(data) {
  indicatorsContainer.innerHTML = "";
  data.forEach(indicator => {
    const card = document.createElement("div");
    card.className = "bg-gray-800 p-4 rounded shadow";
    card.innerHTML = `
      <h2 class="text-xl font-semibold">${indicator.name}</h2>
      <p>${indicator.description}</p>
      <p><strong>Sources:</strong> ${indicator.sources.join(", ")}</p>
    `;
    indicatorsContainer.appendChild(card);
  });
}

searchInput.addEventListener("input", () => {
  const filtered = indicators.filter(i => i.name.toLowerCase().includes(searchInput.value.toLowerCase()));
  renderIndicators(filtered);
});

renderIndicators(indicators);
```
