from flask import Flask, jsonify
import requests
from dotenv import load_dotenv
import os

load_dotenv()
app = Flask(__name__)

@app.route('/api/tweets')
def get_tweets():
    try:
        response = requests.get(
            'https://api.twitter.com/2/tweets/search/recent?query=bitcoin&tweet.fields=created_at,author_id',
            headers={'Authorization': f"Bearer {os.getenv('X_BEARER_TOKEN')}"}
        )
        return jsonify(response.json())
    except Exception as e:
        return jsonify({'error': 'Failed to fetch tweets'}), 500

@app.route('/api/news')
def get_news():
    try:
        response = requests.get(f"https://newsapi.org/v2/everything?q=bitcoin&apiKey={os.getenv('NEWS_API_KEY')}")
        return jsonify(response.json()['articles'])
    except Exception as e:
        return jsonify({'error': 'Failed to fetch news'}), 500

@app.route('/api/grok')
def get_grok():
    try:
        response = requests.get(
            'https://api.x.ai/grok/query?text=bitcoin+insights',
            headers={'Authorization': f"Bearer {os.getenv('GROK_API_KEY')}"}
        )
        return jsonify(response.json())
    except Exception as e:
        return jsonify({'error': 'Failed to fetch Grok insights'}), 500

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=3000)

Add JavaScript for Dynamic Updates (script.js):
Use fetch to call backend endpoints and update the DOM.

Implement polling for continuous updates.
javascript
async function fetchDashboardData() {
  try {
    // Fetch Bitcoin price (e.g., from CoinGecko API for simplicity)
    const priceResponse = await fetch('https://api.coingecko.com/api/v3/simple/price?ids=bitcoin&vs_currencies=usd&include_market_cap=true');
    const priceData = await priceResponse.json();
    document.getElementById('bitcoin-price').textContent = `Price: $${priceData.bitcoin.usd}`;
    document.getElementById('market-cap').textContent = `Market Cap: $${priceData.bitcoin.usd_market_cap}`;
  } catch (error) {
    console.error('Error fetching dashboard data:', error);
  }
}

async function fetchAnalyticsData() {
  try {
    // Fetch sentiment from Grok or X
    const grokResponse = await fetch('/api/grok');
    const grokData = await grokResponse.json();
    document.getElementById('sentiment').textContent = `Sentiment: ${grokData.sentiment || 'Neutral'}`;
  } catch (error) {
    console.error('Error fetching analytics data:', error);
  }
}

async function fetchNewsData() {
  try {
    const newsResponse = await fetch('/api/news');
    const newsData = await newsResponse.json();
    const newsList = document.getElementById('news-list');
    newsList.innerHTML = '';
    newsData.slice(0, 5).forEach(article => {
      const li = document.createElement('li');
      li.innerHTML = `<a href="${article.url}" target="_blank">${article.title}</a> - ${article.source.name}`;
      newsList.appendChild(li);
    });
  } catch (error) {
    console.error('Error fetching news data:', error);
  }
}

// Initial fetch
fetchDashboardData();
fetchAnalyticsData();
fetchNewsData();

// Poll every 5 minutes for dashboard and analytics, every 1 minute for news
setInterval(fetchDashboardData, 300000);
setInterval(fetchAnalyticsData, 300000);
setInterval(fetchNewsData, 60000);
Step 5: Implement Real-Time Updates with WebSocket (Optional)
For truly live updates (e.g., streaming tweets), use WebSocket connections.
Backend WebSocket for X API:

npm install ws
Update server.js:
javascript
const WebSocket = require('ws');
const wss = new WebSocket.Server({ port: 8080 });

wss.on('connection', ws => {
  console.log('Client connected');
  const streamTweets = async () => {
    const response = await axios.get('https://api.twitter.com/2/tweets/sample/stream', {
      headers: { Authorization: `Bearer ${process.env.X_BEARER_TOKEN}` },
      responseType: 'stream'
    });
    response.data.on('data', chunk => {
      const tweet = chunk.toString();
      if (tweet.includes('bitcoin')) {
        ws.send(JSON.stringify({ type: 'tweet', data: tweet }));
      }
    });
  };
  streamTweets();
});

Update server.py:

from websocket_server import WebsocketServer
import requests

def new_client(client, server):
    print("Client connected")
    response = requests.get(
        'https://api.twitter.com/2/tweets/sample/stream',
        headers={'Authorization': f"Bearer {os.getenv('X_BEARER_TOKEN')}"},
        stream=True
    )
    for line in response.iter_lines():
        if line and 'bitcoin' in line.decode('utf-8').lower():
            server.send_message(client, line.decode('utf-8'))

server = WebsocketServer(host='0.0.0.0', port=8080)
server.set_fn_new_client(new_client)
server.run_forever()

Notes
Grok API Limitations: If the Grok API is not publicly available or lacks real-time data endpoints, use Grok’s X integration (e.g., fetch tweets mentioning @Grok with Bitcoin queries). Alternatively, use sentiment analysis libraries like natural (Node.js) or TextBlob (Python) on X data.
X API Costs: The X API’s free tier is limited. You may need a Basic or Pro plan for streaming tweets.
News API: The free newsapi.org plan has a 500-request/day limit. Consider caching or upgrading for heavy usage.

Replit Constraints: Replit’s free tier has resource limits. Upgrade to a paid plan for better performance if needed.

Bitcoin Data: The example uses CoinGecko for price data. Alternatives include CoinMarketCap or Binance APIs.

WebSocket Alternative: If WebSocket setup is complex, stick with polling (e.g., setInterval) for simplicity.
